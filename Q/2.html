    function fibonacci(n) {
        //一、递归解法
        //return n<=2?1:fibonacci(n-1)+fibonacci(n-2);
        
        //二、循环解法
        var num1=1;
        var num2=1;
        for(var i=2;i<n;i++){
                num2+=num1;
                num1=num2-num1;
        }
        return num2;
}
    循环体内的计算过程详解：
    第一句：将num2的旧值，加上num1，得到num2的新值
    第二句：用num2的新值，减去num1，得到num1的新值
    当前循环结束，num1和num2的值都获得了更新~
